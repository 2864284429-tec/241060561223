#### Service 实现要点

**SeatService.GetAllSeatsWithStatus()**

```csharp
public List<SeatListItem> GetAllSeatsWithStatus()
{
    var today = DateTime.Today;
    var currentSlot = DateTime.Now.Hour;

    return _dbContext.Seats
        .Select(seat => new SeatListItem
        {
            Id = seat.Id,
            Name = seat.Name,
            Location = seat.Location,
            Device = seat.Device,
            IsAvailable = seat.IsAvailable,
            // 判断当前时段是否可预约
            CurrentStatus = !seat.IsAvailable ? "不可用" :
                _dbContext.Reservations.Any(r => 
                    r.SeatId == seat.Id && 
                    r.Date == today && 
                    r.TimeSlot == currentSlot && 
                    r.Status == "已预约") ? "已预约" : "可用"
        })
        .ToList();
}
```

**ReservationService.CreateReservation()**

```csharp
public ReservationResult CreateReservation(int seatId, string userId, DateTime date, int timeSlot)
{
    // 1. 检查座位是否存在
    var seat = _dbContext.Seats.Find(seatId);
    if (seat == null)
        return ReservationResult.Fail("座位不存在");

    // 2. 检查座位是否可用
    if (!seat.IsAvailable)
        return ReservationResult.Fail("座位已停用");

    // 3. 检查时段是否已过时
    var slotStartTime = date.AddHours(timeSlot);
    if (DateTime.Now >= slotStartTime)
        return ReservationResult.Fail("该时段已过时");

    // 4. 检查是否已预约（冲突检测）
    var existing = _dbContext.Reservations
        .FirstOrDefault(r => r.SeatId == seatId 
                          && r.Date == date 
                          && r.TimeSlot == timeSlot 
                          && r.Status == "已预约");
    
    if (existing != null)
        return ReservationResult.Fail("该时段已被预约");

    // 5. 创建预约
    var reservation = new Reservation
    {
        SeatId = seatId,
        UserId = userId,
        Date = date,
        TimeSlot = timeSlot,
        Status = "已预约",
        CreatedAt = DateTime.UtcNow,
        UpdatedAt = DateTime.UtcNow
    };

    _dbContext.Reservations.Add(reservation);
    _dbContext.SaveChanges();

    return ReservationResult.Success(reservation.Id);
}
```

### 2.5 链路一：DataAccess 动作

| 步骤 | 数据表 | 操作类型 | SQL 操作 | 说明 |
|------|--------|----------|----------|------|
| 1 | Seats | 查询 | `SELECT * FROM Seats` | 获取座位列表 |
| 2 | Reservations | 查询 | `SELECT * FROM Reservations WHERE Date=@date AND TimeSlot=@slot AND Status='已预约'` | 判断当前时段预约状态 |
| 3 | Seats | 查询 | `SELECT * FROM Seats WHERE Id=@id` | 获取座位详情 |
| 4 | Reservations | 查询 | `SELECT COUNT(*) FROM Reservations WHERE SeatId=@seatId AND Date=@date AND TimeSlot=@slot AND Status='已预约'` | 冲突检测 |
| 5 | Reservations | 插入 | `INSERT INTO Reservations (...) VALUES (...)` | 创建预约记录 |
| 6 | Reservations | 查询 | `SELECT r.*, s.Name FROM Reservations r JOIN Seats s ON r.SeatId = s.Id WHERE r.UserId=@userId` | 查询用户预约列表 |

```csharp
public bool CancelReservation(int reservationId, string userId)
{
    var reservation = _dbContext.Reservations
        .FirstOrDefault(r => r.Id == reservationId && r.UserId == userId);

    // 1. 检查预约是否存在
    if (reservation == null)
        return false;

    // 2. 检查状态是否为"已预约"
    if (reservation.Status != "已预约")
        return false;

    // 3. 检查是否已过时
    var slotEndTime = reservation.Date.AddHours(reservation.TimeSlot + 1);
    if (DateTime.Now >= slotEndTime)
        return false; // 已过时，不能取消

    // 4. 更新状态
    reservation.Status = "已取消";
    reservation.UpdatedAt = DateTime.UtcNow;
    _dbContext.SaveChanges();

    return true;
}
```

### 3.5 异常与边界情况

| 非本人预约 | UserId 不匹配 | 返回 false | "无权操作" |
| 状态异常 | Status != "已预约" | 返回 false | "预约状态异常" |
| 已过时 | 当前时间 >= 时段结束时间 | 返回 false | "已过时，无法取消" |

```csharp
public bool CreateSeat(AdminSeatItem model)
{
    var seat = new Seat
    {
        Name = model.Name,
        Location = model.Location,
        Capacity = model.Capacity,
        Device = model.Device,
        IsAvailable = model.IsAvailable,
        CreatedAt = DateTime.UtcNow,
        UpdatedAt = DateTime.UtcNow
    };

    _dbContext.Seats.Add(seat);
    _dbContext.SaveChanges();
    return true;
}

public bool UpdateSeat(int id, AdminSeatItem model)
{
    var seat = _dbContext.Seats.Find(id);
    if (seat == null) return false;

    seat.Name = model.Name;
    seat.Location = model.Location;
    seat.Capacity = model.Capacity;
    seat.Device = model.Device;
    seat.IsAvailable = model.IsAvailable;
    seat.UpdatedAt = DateTime.UtcNow;

    _dbContext.SaveChanges();
    return true;
}

public bool DeleteSeat(int seatId)
{
    var seat = _dbContext.Seats.Find(seatId);
    if (seat == null) return false;

    // 检查是否有关联预约
    var hasReservations = _dbContext.Reservations
        .Any(r => r.SeatId == seatId && r.Status == "已预约");
    
    if (hasReservations)
        return false; // 有未完成预约，不能删除

    _dbContext.Seats.Remove(seat);
    _dbContext.SaveChanges();
    return true;
}
```
